# Dialog

O componente `Dialog` é reponsável por apresentar uma extensa variedade de conteúdos, seja na de transmissão de informações ao usuário, ou até mesmo na execução de actions.

export const DialogProps = () => {
  const [isOpen, setIsOpen] = useState(false)
  return (
    <>
      <button onClick={() => setIsOpen(true)}>Open Modal</button>
      <Dialog open={isOpen} onClose={() => setIsOpen(false)} closeIcon title="Título" description="Lorem ipsum dolor sit amet, consectetur adipiscing elit.">
        <TextField name='base' label='Default'  placeholder='Default' />
      </Dialog>
    </>
  )
}
